    function clicarBlocoMestre(bloco, elementoDom) {
        // 1. Remove classe ativa dos menus
        document.querySelectorAll('.menu-item').forEach(el => el.classList.remove('active'));
        if (elementoDom) elementoDom.classList.add('active');
        
        // 2. Mostra o Explorer e esconde o Agenda (COM SEGURANÇA)
        let explorer = document.getElementById('viewport-explorer');
        let agenda = document.getElementById('viewport-agenda');
        
        if (explorer) explorer.classList.remove('d-none');
        if (agenda) agenda.classList.add('d-none'); // Só esconde se existir

    function abrirConfiguradorAgenda() {
        // Removemos o gerenciamento de viewport que estava escondendo o modal

    function alternarCamposAgenda(tipo) {
        if(tipo === 'reuniao') {
            document.getElementById('grupo-campos-reuniao').classList.remove('d-none');
            document.getElementById('grupo-campos-evento').classList.add('d-none');
        } else {
            document.getElementById('grupo-campos-reuniao').classList.add('d-none');
            document.getElementById('grupo-campos-evento').classList.remove('d-none');
        }
    }

    function salvarEventoFormulario() {
        let t = document.getElementById('tituloEvento').value.trim();
        let tipo = document.getElementById('tipoEventoSelect').value;
        if(!t) return alert("Por favor, digite o título do compromisso!");

        let fd = new URLSearchParams();
        fd.append('titulo', t);
        fd.append('tipo_evento', tipo);

        if(tipo === 'reuniao') {
            let data = document.getElementById('dataReuniaoInput').value;
            let hora = document.getElementById('horarioReuniaoInput').value;
            if(!data) return alert("Selecione a data!");
            fd.append('data_inicio', data);
            fd.append('data_fim', '');
            fd.append('horario', hora);
        } else {
            let dataIni = document.getElementById('dataInicioEventoInput').value;
            let dataFim = document.getElementById('dataFimEventoInput').value;
            if(!dataIni || !dataFim) return alert("Preencha o período completo do evento!");
            fd.append('data_inicio', dataIni);
            fd.append('data_fim', dataFim);
            fd.append('horario', '00:00');
        }
        
        fetch('/adicionar-evento', { method: 'POST', body: fd })
        .then(() => {
            bootstrap.Modal.getInstance(document.getElementById('eventoModal')).hide();
            if(instanciaCalendario) { instanciaCalendario.refetchEvents(); }
            alert("Compromisso inserido com sucesso na sua Agenda!");
        });
    }

function sincronizarGrade() {
    let gridContainer = document.getElementById("wrapper-explorer-grid");
    let caminhoEl = document.getElementById('txtCaminhoWindows');
    
    // Identifica os grupos de botões no HTML
    let divArquivos = document.getElementById('controles-arquivos');
    let divSites = document.getElementById('controles-sites');
    
    // Procura os botões na página baseando-se no texto que eles exibem atualmente
    let botoes = document.querySelectorAll('button');
    let btnNovaPasta = null;
    let btnEnviarArquivo = null;

    botoes.forEach(btn => {
        // Encontramos os botões pelo texto atual (identificando qual é qual)
        if(btn.innerText.includes('Nova Pasta') || btn.innerText.includes('Incluir Site')) btnNovaPasta = btn;
        if(btn.innerText.includes('Enviar Arquivo') || btn.innerText.includes('Incluir Logo')) btnEnviarArquivo = btn;
    });

    // Lógica de alteração dinâmica dos nomes dos botões E visibilidade das DIVs
    if (estadoDoPainel.bloco === 'sites_jp2') {
        // Módulo Sites: mostra o grupo de sites e esconde o de arquivos
        if(divArquivos) divArquivos.classList.add('d-none');
        if(divSites) divSites.classList.remove('d-none');

        // Atualiza os nomes para os botões do módulo de Sites
        if (btnNovaPasta) btnNovaPasta.innerHTML = '<i class="bi bi-plus-circle"></i> Incluir Site';
        if (btnEnviarArquivo) btnEnviarArquivo.innerHTML = '<i class="bi bi-image"></i> Incluir Logo';
    } else {
        // Módulo Arquivos (Instituto/JP2): mostra o grupo de arquivos e esconde o de sites
        if(divArquivos) divArquivos.classList.remove('d-none');
        if(divSites) divSites.classList.add('d-none');

        // Restaura os nomes originais para os módulos de arquivos
        if (btnNovaPasta) btnNovaPasta.innerHTML = '<i class="bi bi-folder-plus"></i> Nova Pasta';
        if (btnEnviarArquivo) btnEnviarArquivo.innerHTML = '<i class="bi bi-cloud-arrow-up"></i> Enviar Arquivo';
    }

    if (!gridContainer || !caminhoEl) return;

    caminhoEl.innerText = estadoDoPainel.string_caminho;
    gridContainer.innerHTML = "";

    if (estadoDoPainel.nivel_atual === 'raiz_bloco') {
        let btnVoltar = document.getElementById('btnVoltarExplorer');
        if(btnVoltar) btnVoltar.style.visibility = 'hidden';
        
        let categorias = subpastasFixas[estadoDoPainel.bloco];
        categorias.forEach(cat => {
            let card = document.createElement("div");
            card.className = "explorer-card";
            
            if (estadoDoPainel.bloco === 'sites_jp2') {
                let urlDestino = linksDirecionamento[cat] || '#';
                let nomeArquivo = mapeamentoLogos[cat] || 'default.png';
                
                card.onclick = function() { window.open(urlDestino, '_blank'); };
                card.innerHTML = `
                    <div class="icon-wrapper">
                        <img src="/static/image/${nomeArquivo}" alt="${cat}" style="width: 50px; height: 50px; object-fit: contain; border-radius: 8px;">
                    </div>
                    <div class="item-label">${cat}</div>
                    <div class="item-author">Portal WEB</div>
                `;
            } else {
                card.onclick = function() { entrarNaCategoriaFixa(cat); };
                card.innerHTML = `<div class="icon-wrapper"><i class="bi bi-folder-fill text-warning"></i></div><div class="item-label">${cat}</div><div class="item-author">Diretório</div>`;
            }
            gridContainer.appendChild(card);
        });
        
        fetch(`/listar?bloco=${estadoDoPainel.bloco}&categoria=raiz&pasta_pai_id=null`)
        .then(res => res.json())
        .then(data => {
            if(data && data.itens) {
                data.itens.forEach(item => {
                    if(item.tipo === 'pasta') {
                        let card = document.createElement("div");
                        card.className = "explorer-card";
                        card.innerHTML = `<div class="action-trigger"><button class="btn btn-sm btn-light border py-0 px-1 text-danger" onclick="abrirModalExcluirDireto(event, ${item.id})"><i class="bi bi-trash"></i></button></div><div class="icon-wrapper" onclick="entrarSubpastaReal(${item.id}, '${item.nome}')"><i class="bi bi-folder-fill text-warning"></i></div><div class="item-label" onclick="entrarSubpastaReal(${item.id}, '${item.nome}')">${item.nome}</div>`;
                        gridContainer.appendChild(card);
                    }
                });
            }
        });
    } else {
        let btnVoltar = document.getElementById('btnVoltarExplorer');
        let btnCriar = document.getElementById('btnCriarPastaExplorer');
        let btnUpload = document.getElementById('btnNovoItemDrive');
        
        if(btnVoltar) btnVoltar.style.visibility = 'visible';
        if(btnCriar) btnCriar.style.visibility = 'visible';
        if(btnUpload) btnUpload.removeAttribute('disabled');
        
        gridContainer.innerHTML = '<div class="text-center w-100 py-4 text-muted"><div class="spinner-border spinner-border-sm text-primary mb-2"></div><br>Buscando na nuvem...</div>';
        
        fetch(`/listar?bloco=${estadoDoPainel.bloco}&categoria=${estadoDoPainel.categoria}&pasta_pai_id=${estadoDoPainel.pasta_pai_id}`)
        .then(res => res.json())
        .then(data => {
            gridContainer.innerHTML = "";
            if (data.itens.length === 0) {
                gridContainer.innerHTML = '<div class="text-center w-100 py-4 text-muted"><i class="bi bi-folder-x fs-3"></i><br>Nenhum item ou subpasta aqui.</div>';
                return;
            }
            data.itens.forEach(item => {
                let card = document.createElement("div");
                card.className = "explorer-card";
                let iconHtml = item.tipo === 'pasta' ? '<i class="bi bi-folder-fill text-warning"></i>' : '<i class="bi bi-file-earmark-text-fill text-primary"></i>';
                let cliqueAcao = item.tipo === 'pasta' ? `onclick="entrarSubpastaReal(${item.id}, '${item.nome}')"` : `onclick="window.open('/baixar/${item.id}', '_blank')"`;
                card.innerHTML = `
                    <div class="action-trigger">
                        <button class="btn btn-sm btn-light border py-0 px-1 text-danger" onclick="abrirModalExcluirDireto(event, ${item.id})"><i class="bi bi-trash"></i></button>
                    </div>
                    <div class="icon-wrapper" ${cliqueAcao}>${iconHtml}</div>
                    <div class="item-label" ${cliqueAcao} title="${item.nome}">${item.nome}</div>
                    <div class="item-author">${item.autor}</div>
                `;
                gridContainer.appendChild(card);
            });
        });
    }
}

    function entrarNaCategoriaFixa(nomeCategoria) {
        estadoDoPainel.categoria = nomeCategoria;
        estadoDoPainel.nivel_atual = 'subpasta';
        estadoDoPainel.pasta_pai_id = null;

    function entrarSubpastaReal(id, nomePasta) {

        estadoDoPainel.nivel_atual = 'subpasta';

    }

    function voltarDiretorio() {
        if (estadoDoPainel.pasta_pai_id !== null) {
            estadoDoPainel.pasta_pai_id = null;
            let partes = estadoDoPainel.string_caminho.split(' ➔ ');
            estadoDoPainel.string_caminho = partes[0] + ' ➔ ' + partes[1];
            sincronizarGrade();
        } else if (estadoDoPainel.nivel_atual === 'subpasta') {
            estadoDoPainel.nivel_atual = 'raiz_bloco';
            estadoDoPainel.categoria = '';
            estadoDoPainel.string_caminho = estadoDoPainel.bloco.toUpperCase().replace('_', ' ');
            sincronizarGrade();
        }
    }

            headers: {'Content-Type': 'application/x-www-form-urlencoded'},
            body: `nome=${encodeURIComponent(nome)}&bloco=${encodeURIComponent(estadoDoPainel.bloco)}` 
        })

            // Se o status não for 200, lemos o erro como texto em vez de JSON
            if (!response.ok) {

            }
            return response.json();
        })

            alert("Pasta criada!");
            overlay.remove();
            setTimeout(() => { sincronizarGrade(); }, 500);
        })
        .catch(err => {
            console.error("Erro detalhado do servidor:", err);
            alert("Erro no servidor! Verifique o log do seu terminal Python.");
        });
    };
