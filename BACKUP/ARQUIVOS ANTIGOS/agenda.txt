    <div class="agenda-fullscreen-container" style="background: white; padding: 25px; border-radius: 12px; border: 1px solid #e2e8f0;">
        <div class="d-flex justify-content-between align-items-center mb-4">
            <h2>Minha Agenda - JP2BUSINESS</h2>
        </div>

        <div id='calendar'></div>
    </div>

    <script>
        document.addEventListener('DOMContentLoaded', function() {
            var calendarEl = document.getElementById('calendar');
            
            var calendar = new FullCalendar.Calendar(calendarEl, {
                initialView: 'dayGridMonth',
                locale: 'pt-br',
                events: '/api/listar-eventos',
                dayMaxEvents: false, 
                displayEventTime: false,
                
                eventContent: function(arg) {
                    return {
                        html: '<div class="fc-event-title">' + arg.event.title + '</div>'
                    };
                },
                
                dateClick: function(info) {
                    abrirConfiguradorAgenda();
                },
                
                eventClick: function(info) {
                    if (confirm('Deseja excluir o compromisso: ' + info.event.title + '?')) {
                        fetch('/api/excluir-evento', {
                            method: 'POST',
                            headers: { 'Content-Type': 'application/json' },
                            body: JSON.stringify({ 'titulo': info.event.title })
                        }).then(() => {
                            info.event.remove();
                        });
                    }
                }
            });

            calendar.render();
            window.instanciaCalendario = calendar;
        });
    </script>
{% endblock %}
