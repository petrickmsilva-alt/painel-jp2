        document.body.appendChild(btnVoltar);

        .then(data => {

                document.getElementById('dash-arquivos').innerText = data.total_arquivos;
                document.getElementById('dash-socios').innerText = data.total_socios;
                document.getElementById('dash-agenda').innerText = data.proximo_evento;
            }
        });
    });
